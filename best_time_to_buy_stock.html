Best Time to Buy and Sell Stock

        
            class Solution(object):
                def maxProfit(self, prices):
                    """
                    :type prices: List[int]
                    :rtype: int
                    """
                    ans, carry = 0, prices[0]
                    for i in range(1, len(prices)):
                        carry = min(carry, prices[i])
                        ans = max(ans, prices[i] - carry)
                    return ans

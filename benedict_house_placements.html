Count House Placements

Find the best way to rob the neighborhood without getting arrested.

        
            class Solution(object):
                def countHousePlacements(self, n):
                    """
                    :type n: int
                    :rtype: int
                    """
                    a, b = 1, 2

                    for i in range(1, n):
                        a, b = b, a + b

                    return (b ** 2) % (10 ** 9 + 7)

        def main():
            dollars = dollars_to_float(input("How much was the meal? "))

            tip = dollars * percent

        #turns dollars to float. removes the dollar sign first
        def dollars_to_float(d):
            return float(d[1:len(d)])

        #removes the percent sign and divides by 100 to get percentage in decimal
        def percent_to_float(p):
            return float(p[:-1])/100


        main()
